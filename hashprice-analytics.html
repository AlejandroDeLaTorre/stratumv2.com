// DATABASE SCHEMA
-- Core snapshot table
CREATE TABLE hashprice_snapshots (
  id          BIGSERIAL PRIMARY KEY,
  timestamp   TIMESTAMPTZ NOT NULL,
  price_usd   NUMERIC(18,8) NOT NULL,
  price_btc   NUMERIC(18,12) NOT NULL,
  hash_unit   VARCHAR(4) DEFAULT 'THS',
  created_at  TIMESTAMPTZ DEFAULT now()
);

-- Index for fast time-range queries
CREATE INDEX idx_ts
  ON hashprice_snapshots (timestamp DESC);

-- Daily OHLC aggregates
CREATE TABLE hashprice_daily (
  date        DATE PRIMARY KEY,
  open_usd    NUMERIC(18,8),
  high_usd    NUMERIC(18,8),
  low_usd     NUMERIC(18,8),
  close_usd   NUMERIC(18,8),
  avg_btc     NUMERIC(18,12)
);
